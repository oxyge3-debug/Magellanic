stargen
    var loadingamount = 1
    repeat sphere(var(Li), location(25,57,25), %random(2,4), 600)
    var(Li) randomizedsurface = %random(2,4)

    if var(Li) randomizedsurface < 2
        shiftindirection
            var(Li) loc =
            var(Li) loc
            num -0.3
        appendvalue
            list(G) land
            var(Li) loc
    else if var(Li) randomizedsurface >= 2
        appendvallue
            list(G) water
            var(Li) loc

    loadingamount += 1
    styledtext
        var(G) loading =
        str "loading"
        var(G) loadingamount
        str "/600"

    actionbar(allplayers)
        var(G) loading

    wait
